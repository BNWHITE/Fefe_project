      \item Contr\^ole vocal mains-libres
      \item Authentification Google OAuth
    \end{itemize}
  \end{columns}

  \end{center}

\begin{frame}{Flux de Donn\'ees}
  \begin{enumerate}
    \item \textbf{Capteurs IoT} $\rightarrow$ Mesures (temp\'erature, humidit\'e, luminosit\'e)
    \item \textbf{LiDAR G2D} $\rightarrow$ Distance (mm), r\'eflectivit\'e (\%), luminosit\'e (lux)
    \item \textbf{MySQL} $\rightarrow$ Stockage temps r\'eel (AlwaysData)
    \item \textbf{PHP API} $\rightarrow$ Exposition REST des donn\'ees
    \item \textbf{React Dashboard} $\rightarrow$ Affichage live (200 ms refresh)
    \item \textbf{IA Groq/Mistral} $\rightarrow$ Analyse contextuelle + conseils
  \end{enumerate}

  \vspace{0.3cm}
  \begin{block}{Points cl\'es}
    \begin{itemize}
      \item 3 niveaux de fallback IA : Groq $\rightarrow$ Mistral $\rightarrow$ FAQ BDD $\rightarrow$ Local
      \item Proxy Vite PHP/Node pour le d\'eveloppement local
      \item D\'eploiement Vercel pour la production
    \end{itemize}
  \end{block}
\end{frame}

\section{S\'ecurit\'e \& Authentification}
% ============================================================

\begin{frame}{Google OAuth 2.0}
  \begin{columns}
    \column{0.55\textwidth}
    \textbf{\textcolor{ferrariRed}{Flux d\'authentification :}} \\[0.2cm]
    \begin{enumerate}
      \item Utilisateur clique « Google Sign-In »
      \item Google v\'erifie l\'identit\'e
      \item JWT d\'ecod\'e c\^ot\'e client
      \item Session persist\'ee (localStorage)
      \item Dashboard prot\'eg\'e par \texttt{useAuth()}
    \end{enumerate}

    \column{0.45\textwidth}
    \textbf{\textcolor{ferrariRed}{S\'ecurit\'e :}} \\[0.2cm]
    \begin{itemize}
      \item Cl\'es API \textbf{jamais} expos\'ees
      \item \texttt{.env.local} exclu du Git
      \item Rate limiting (10 req/min par IP)
      \item Fallback local FERRARI/16
      \item CORS headers sur l\'API
    \end{itemize}
  \end{columns}

  \vspace{0.4cm}
  \begin{alertblock}{Note}
    Les cl\'es Groq/Mistral sont stock\'ees c\^ot\'e serveur (PHP) et \textbf{jamais} envoy\'ees au client.
  \end{alertblock}
\end{frame}
